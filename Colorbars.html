Colormap From List¶

This makes a colormap more similar to how Matlab works. Some of what is done here can also be done directly in the colormaps, but this is just a demo

In [9]:
# Make a list with colors
nc = 500
x = np.linspace(0,1,nc)

c0 = [[0.36392156862745095, 0.5755294117647058, 0.7483921568627451],
      [0.91529411764705880, 0.2815686274509803, 0.2878431372549019],
      [0.44156862745098036, 0.7490196078431373, 0.4321568627450980]]
c1 = [np.interp(x,np.linspace(0,1,len(c0)),c) for c in np.atleast_2d(c0).T]
c1 = np.vstack(c1).T

cmap = mpl.colors.ListedColormap(c1)

############### Plot it
# create a dummy plot
_f,_a = plt.subplots()
pl = plt.imshow(np.random.uniform(size=(4,4)),vmin=0,vmax=15,cmap=cmap)
plt.close(_f)

fig,ax = plt.subplots(figsize=(5,0.8))
cbar = fig.colorbar(pl,cax=ax,orientation='horizontal')
fig.tight_layout()
cbar.set_ticks([0,5,10,15])

A helpful function¶

def pretty_colorbar(pcm_obj,fig=None,ax=None,size="5%",pad=0.05):
    """
    Create a nicer color bar on the side of the plot
    
    Inputs:
        pcm_obj : the  `ax.pcolormesh` object
        fig     :   [`plt.gcf()`] The matplotlib figure
        ax      :   [`plt.gca()`] The matplotlib axis
        size    :   ["5%"] size of the colorbar
        pad     :   [0.05] The padding
    
    Returns:
        cbar    :   fig.colorbar object
        
    Tips:
        Set the scale in the `pcolormesh` but you can set the ticks with:
            
            cbar.set_ticks(np.linspace(-1,2.2,6))
    """
    if fig is None:
        fig = plt.gcf()
    if ax is None:
        ax = plt.gca()
    
    from mpl_toolkits.axes_grid1 import make_axes_locatable
    divider = make_axes_locatable(ax)
    cax = divider.append_axes("right", size=size, pad=pad)
    cbar = fig.colorbar(pcm_obj,cax=cax)
    return cbar
